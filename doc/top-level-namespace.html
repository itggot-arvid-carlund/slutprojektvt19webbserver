post('/login') do
    if login_(params["username"], params["password"]) == true
        session[:logged_in] = true
        session[:username] = params["username"]
        session[:userID] = get_userID(params["username"])
        redirect('/welcome')
    else
        session[:logged_in] = false
        redirect('/')
    end    
end
